// Саставити програм који учитава, а затим исписује елементе матрице mxn.
// Елементи матрице су цели бројеви.

#include <stdio.h>
#define MAX 100 // Максимална величина матрице

int main()
{
    int i, j, m, n, mat[MAX][MAX]; // Декларација променљивих

    // Учитавање броја редова и колона
    printf(" Vrsta m = ");
    scanf("%d", &m);
    printf(" Kolona n = ");
    scanf("%d", &n);

    // Учитавање елемената матрице
    for (i = 0; i < m; i++)
        for (j = 0; j < n; j++)
        {
            printf(" element[%d][%d] = ", i, j); // Тражи унос елемента
            scanf("%d", &mat[i][j]);
        }

    printf("\n Uneta matrica: ");
    // Испис елемената матрице
    for (i = 0; i < m; i++)
    {
        for (j = 0; j < n; j++)
            printf(" %d", mat[i][j]); // Исписује елемент
        printf("\n");                 // Прелазак у нови ред након сваког реда
    }

    getchar(); // Чека унос корисника пре завршетка
    return 0;  // Завршетак програма
}
