// б) чита са тастатуре неограничени број каректера све док се не учита EOF  и исписује унете карактере на екрану.

#include <stdio.h>

int main()
{
    int c = 0; // Иницијализујемо c на неку вредност
    while (c != EOF)
    {
        c = getchar();
        putchar(c);
    }

    getchar();
    return 0;
}
