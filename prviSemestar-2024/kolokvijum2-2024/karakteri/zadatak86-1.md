// Задатак 86:
// Саставити програм који врши конверзију унетих великих слова у мала.
// Унос текста се завршава сигналом ЕОF.

// Први начин (тежи):

#include <stdio.h>

int main()
{
    int c;

    while ((c = getchar()) != EOF)
    {
        if (c >= 'A' && c <= 'Z')
            c = c - 'A' + 'a';
        putchar(c);
    }

    getchar();
    return 0;
}
