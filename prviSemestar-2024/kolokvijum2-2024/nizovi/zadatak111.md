// Саставити програм који за унети низ (дужине n) целих бројева израчунава
// и исписује аритметичку средину оних елемената низа који су дељиви са бројем 3.

#include <stdio.h>
#define MAX 100 // Максимална величина низа

int main()
{
    int i, n, k = 0, niz[MAX];
    double suma = 0;

    // Унос броја елемената низа
    printf(" Broj elemenata: ");
    scanf("%d", &n);

    // Унос елемената низа
    printf(" Elementi niza: ");
    for (i = 0; i < n; i++)
    {
        scanf("%d", &niz[i]);

        // Провера да ли је елемент дељив са 3
        if (niz[i] % 3 == 0)
        {
            suma += niz[i]; // Додавање елемента на суму
            k++;            // Увећање бројача елемената дељивих са 3
        }
    }

    // Испис аритметичке средине елемената дељивих са 3
    printf(" SRV = %.3f\n", suma / k);

    getchar(); // Чека унос корисника пре завршетка
    return 0;  // Завршетак програма
}
