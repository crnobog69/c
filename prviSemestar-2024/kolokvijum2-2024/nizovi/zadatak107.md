// Шта се исписује на екрану након извршавања следећег програма:

#include <stdio.h>
#define MAX 5 // Дефинишемо MAX као 5, што ће бити величина низа

int main()
{
    int i, niz[] = {15, 3, 9, 7, 11}; // Декларишемо низ са 5 елемената

    // Исписујемо заглавље табеле на екрану
    printf("\nR.br.\tIndeks\tVrednost\n");

    // Петља која пролази кроз све елементе низа
    for (i = 0; i < MAX; i++)
        // Исписујемо број реда, индекс елемента и саму вредност елемента
        printf("\n%d\t%d\t%d", i + 1, i, niz[i]);

    getchar(); // Чекамо да корисник притисне било који тастер пре него што програм заврши
    return 0;  // Завршавамо програм
}
