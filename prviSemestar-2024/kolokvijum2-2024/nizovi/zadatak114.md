```c
#include <stdio.h>
#include <math.h>
#define MAX 100

int main()
{
    float nizA[MAX], nizB[MAX], nizC[MAX]; // Декларација три низа са максималном величином MAX
    int i, n;                              // Декларација променљивих за итерацију и број елемената низа

    printf(" n = "); // Испис поруке за унос броја елемената низа
    scanf("%d", &n); // Унос броја елемената низа

    printf(" Елементи низа A: "); // Испис поруке за унос елемената низа A
    for (i = 0; i < n; i++)       // Петља за унос елемената низа A
        scanf("%f", &nizA[i]);    // Унос елемената низа A

    printf(" Елементи низа B: "); // Испис поруке за унос елемената низа B
    for (i = 0; i < n; i++)       // Петља за унос елемената низа B
        scanf("%f", &nizB[i]);    // Унос елемената низа B

    printf("\n Елементи низа C: "); // Испис поруке за испис елемената низа C
    for (i = 0; i < n; i++)         // Петља за израчунавање и испис елемената низа C
    {
        // Израчунавање елемената низа C према формули
        nizC[i] = pow(nizA[i], 3) / 3 + 2 * nizA[i] * nizB[i];
        printf("%.2f ", nizC[i]); // Испис елемената низа C са две децимале
    }

    printf("\n"); // Празна линија за испис резултата

    getchar(); // Чекање на унос карактера пре завршетка програма
    return 0;  // Повратна вредност функције main
}
```
