```c
#include <stdio.h> // Ово нам омогућава да користимо функције за унос и износ података

int main() // Овде почиње наш главни програм
{
    int n, i = 1, suma = 0; // Декларишемо три променљиве:
                            // n - број који уносимо
                            // i - почињемо од 1 и користимо за бројач
                            // suma - овде ћемо чувати збир

    printf("Unesite prirodan broj: "); // Овде питамо корисника да унесе број
    scanf("%d", &n);                   // Овде узимамо број који је корисник унео

    // Овде почињемо циклус који ће се извршавати све док је i мање или једнако n
    while (i <= n)
    {
        suma = suma + i; // Додајемо i на збир
        i = i + 3;       // Повећавамо i за 3 (три корака)
    }

    // Овде исписујемо резултат
    printf("Suma svakog treceg broja, od 1 do %d, je %d\n", n, suma);

    return 0; // Завршавамо програм
}
```
